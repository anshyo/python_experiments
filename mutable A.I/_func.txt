def func_editor(changing , change):
    func_content = ''
    func_file = open('mutable A.I\_func.py' , 'r')
    c = 0
    for line in func_file:
        if changing in line:
            if c == 1:
                break
            func_content += line.replace(changing , change)
            c += 1
        else:
            func_content += line
    func_file.close()
    func_file = open('mutable A.I\_func.py' , 'w')
    func_file.write(func_content)
    func_file.close()
